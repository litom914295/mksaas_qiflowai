import axios, { AxiosRequestConfig } from 'axios';
import pTimeout from 'p-timeout';


export async function httpPostWithRetry(
  url: string,
  body: any,
  timeout = 5000,
  maxRetries = 2
): Promise<{ ok: boolean; data?: any; error?: string }> {
  let lastErr: any = null;
  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      const cfg: AxiosRequestConfig = { headers: { 'Content-Type': 'application/json' }, timeout };
      const res = await axios.post(url, body, cfg);
      return { ok: true, data: res.data };
    } catch (err: any) {
      lastErr = err;
      const backoff = 200 * Math.pow(2, attempt);
      await new Promise((r) => setTimeout(r, backoff));
    }
  }
  return { ok: false, error: String(lastErr?.message || lastErr) };
}

async function callDeterministicServices(input: ValidatedInput) {
  const tasks: Record<string, Promise<ServiceResult>> = {
    bazi: httpPostWithRetry(BAZI_URL, { birth: input.birth, timezone: input.timezone }, DEFAULT_TIMEOUT, MAX_RETRIES),
    fengshui: httpPostWithRetry(FENGSHUI_URL, { orientation: input.orientation, build_year: input.build_year }, DEFAULT_TIMEOUT, MAX_RETRIES),
    calendar: httpPostWithRetry(CAL_URL, { birth: input.birth }, DEFAULT_TIMEOUT, MAX_RETRIES),
    geo: httpPostWithRetry(GEO_URL, { latitude: input.latitude, longitude: input.longitude }, DEFAULT_TIMEOUT, MAX_RETRIES)
  };


  const keys = Object.keys(tasks);
  const settled = await Promise.allSettled(Object.values(tasks));


  const mapped: Record<string, ServiceResult> = {};
  for (let i = 0; i < keys.length; i++) {
    const k = keys[i];
    const r = settled[i];
    if (r.status === 'fulfilled') mapped[k] = r.value;
    else mapped[k] = { ok: false, error: String(r.reason) };
  }
  return mapped;
}


function aggregateResults(input: ValidatedInput, services: Record<string, ServiceResult>) {
  return {
    request_id: input.request_id,
    input,
    bazi: services.bazi?.ok ? services.bazi.data : null,
    fengshui: services.fengshui?.ok ? services.fengshui.data : null,
    calendar: services.calendar?.ok ? services.calendar.data : null,
    geo: services.geo?.ok ? services.geo.data : null,
    metadata: {
      fetched_at: new Date().toISOString(),
      errors: Object.fromEntries(Object.entries(services).filter(([k, v]) => !v.ok).map(([k, v]) => [k, v.error]))
    }
  };
}
